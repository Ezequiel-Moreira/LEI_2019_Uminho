\section{Definition of Derivative}
\begin{defi}
A function $f:\R \to \R$ is differentiable  at $x \in \R$, if there exists a number $f $$'$ $(x)$ such that:
\begin{align*}
\lim_{\varepsilon \to 0} \frac{\textit{$f$($x$ $+$ $\varepsilon$) - $f(x)$}}{\varepsilon} \ = \textit{$f $$'$ $(x)$}  
\end{align*}

-- -- Pause -- --

\begin{align*}
\hspace{-2mm} \lim_{\varepsilon \to 0} \frac{\textit{$f$($x$ $+$ $\varepsilon$) - $f(x)$}}{\varepsilon} - \textit{$f $$'$ $(x)$} = 0 \hspace{0.5mm} 
\Leftrightarrow \hspace{0.5mm} \lim_{\varepsilon \to 0} \frac{\textit{$f$($x$ $+$ $\varepsilon$) - $f(x)$} - (\varepsilon \cdotp \textit{$f $$'$ $(x)$})}{\varepsilon} = 0 
\end{align*}

-- -- Pause -- --

\end{defi}

\begin{defi}
A function $f:\R^{m} \to \R^{n}$ is differentiable  at $x \in \R^{m}$, if there exists a unique linear transformation $\lambda:\R^{m} \to \R^{n}$  such that:
\begin{align*}
\lim_{\varepsilon \to 0} \frac{ \| \textit{$f$($x$ $+$ $\varepsilon$) - $f(x)$} - \lambda (\varepsilon)  \|}{\| \varepsilon \|} = 0
\end{align*}
\end{defi}
